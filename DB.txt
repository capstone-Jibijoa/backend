CREATE TABLE welcome_meta (
    pid SERIAL PRIMARY KEY,                  -- 내부 식별자 (자동 증가)
    panel_id_raw VARCHAR(50) UNIQUE NOT NULL, -- 패널 고유 ID (중복 방지)
    structured_data JSONB NOT NULL,           -- 설문 및 패널 데이터 전체 JSON
    created_at TIMESTAMP DEFAULT NOW(),       -- 생성 시각
    updated_at TIMESTAMP DEFAULT NOW()        -- 수정 시각
);

CREATE TABLE search_log (
    id BIGSERIAL PRIMARY KEY,
    query TEXT NOT NULL,
    results_count INTEGER,
    uid INTEGER,
    query_type_info JSONB,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
